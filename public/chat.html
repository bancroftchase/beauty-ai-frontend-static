const BACKEND_URL = "https://beauty-ai-backend.onrender.com"; async function sendMessage() { const input = document.getElementById('messageInput'); const message = input.value.trim(); if (!message) return; input.value = ''; clearProducts(); appendMessage('user', message); appendMessage('bot', 'Searching global beauty database...'); try { const response = await fetch(`${BACKEND_URL}/search-beauty-products`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }, body: JSON.stringify({ category: message }) }); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); if (data.error) { updateLastBotMessage(`Error: ${data.error}`); return; } if (!data.length) { updateLastBotMessage('No products found. Try different terms like "tanning lotions" or "K-beauty serums"'); return; } products = data; currentIndex = 0; showNextBatch(); } catch (error) { console.error('Fetch error:', error); updateLastBotMessage('Failed to connect to beauty database. Please try again later.'); } }
